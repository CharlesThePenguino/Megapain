namespace = alex_andalus
country_event = { 
	id = alex_andalus.1
	title = alex_andalus.1.t
	desc = alex_andalus.1.d
	picture = PRAYING_eventPicture
	
	trigger = {
	    tag = ADU
		NOT = { has_global_flag = alex_andalus}	 
		num_of_owned_provinces_with = {
				is_core = ADU
				region = iberia_region
				value = 62
				religion = sunni
			}
			iberia_region = {
				type = all
				owner = { religion_group = muslim }
			}
			
			}
	
	mean_time_to_happen = { 
		days = 2
	}
	immediate = {
		set_global_flag = alex_andalus
		play_sound = alex_andalus
	}
	fire_only_once = yes
	option = { 
		name = alex_andalus.1.a		
		custom_tooltip = "Can we book tickets?"
		every_country = {
		limit = { ai = no }
        country_event = {
        id = alex_andalus.2
        }
		}
	}
}
